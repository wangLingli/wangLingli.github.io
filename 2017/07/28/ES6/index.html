let 和 const

let 的作用和 var 一样，都是用来声明变量，但是 let 只在代码块中有效。

function () {
let a = 1;
var a = 2;
}
// Identifier ‘a’ has already been declared

let arr = [1, 2, 3, 4, 5];
let a = arr[0];
let b = arr[1];
let c = arr[2];
let d = arr[3];
let e = arr[4];

let arr = [1, 2, 3, 4, 5];
let [a, b, c, d, e] = arr;

let arr = [1, 2, [3, 4, 5]];
let [a, b, [c, d, e]] = arr;

let [a, b, c] = [1, 2];
// a 1
// b 2
// c undefined

let [a, [b], d] = [1, [2, 3], 4];
a // 1
b // 2
d // 4

let [x, y = ‘b’] = [‘a’]; // x=’a’, y=’b’

let { bar, foo } = { foo: “aaa”, bar: “bbb” };
foo // “aaa”
bar // “bbb”

let [a,b,c,d,e] = ‘hello’;
let {length:len} = ‘hello’;
a //‘h’
len //5

//以前我们要写一个函数并具有返回值：
let fn = function() { return 5 };

//ES6
let fn = () => 5;

let sum = (n1, n2) => { return n1 + n2; }

let fn = () => ({ id: “n1”, name: “aaa” });

var obj = {
x : 1,
func : function() { console.log(this.x) },
test : function() {
setTimeout(function() {
this.func();
}, 100);
}
};

obj.test(); // TypeError : this.func is not a function

var obj = {

setTimeout(() => { this.func() }, 100);

};

obj.test(); // 1

class Animal {
constructor(){
this.type = ‘animal’
}
says(say){
console.log(this.type + ‘ says ‘ + say)
}
}

let animal = new Animal()
animal.says(‘hello’) //animal says hello

class Cat extends Animal {
constructor(){
super()
this.type = ‘cat’
}
}

let cat = new Cat()
cat.says(‘hello’) //cat says hello
